238         private void shareHandler(boolean stub){
239             String shareString = user.getName()+" attended the game between the "+event.getHomeTeam()+" and the "+event.getRoadTeam()+" at "+event.getStadium().getName()+" on "+event.getDateFullString()+". ";
240             if (event.getHomeScore() > event.getRoadScore()){
241                 shareString += "The "+event.getHomeTeam().getNickname()+" won the game by a score of "+event.getHomeScore()+" - "+event.getRoadScore();
242             }else if (event.getRoadScore() > event.getHomeScore()){
243                 shareString += "The "+event.getRoadTeam().getNickname()+" won the game by a score of "+event.getRoadScore()+" - "+event.getHomeScore();
244             }else{
245                 shareString += "The game resulted in a tie with a score of "+event.getHomeScore();
246             }
247     
248             if (stub){
249                 Bitmap ticket = generateStub();
250                 String state = Environment.getExternalStorageState();
251                 if (!Environment.MEDIA_MOUNTED.equals(state)) {
252                     Toast.makeText(this, "Cannot Access External Storage at this time", Toast.LENGTH_SHORT).show();
253                 }
254                 Uri uri = saveBitmap(ticket);
255                 Intent sendIntent = new Intent();
256                 sendIntent.setAction(Intent.ACTION_SEND);
257                 sendIntent.putExtra(Intent.EXTRA_TEXT, shareString);
258                 sendIntent.putExtra(Intent.EXTRA_STREAM, uri);
259                 sendIntent.addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION);
260                 sendIntent.setType("*/*");
261                 Intent shareIntent = Intent.createChooser(sendIntent, null);
262                 startActivity(shareIntent);
263             }else {
264                 Intent sendIntent = new Intent();
265                 sendIntent.setAction(Intent.ACTION_SEND);
266                 sendIntent.putExtra(Intent.EXTRA_TEXT, shareString);
267                 sendIntent.setType("text/plain");
268                 Intent shareIntent = Intent.createChooser(sendIntent, null);
269                 startActivity(shareIntent);
270             }
271         }

349         private Uri saveBitmap(Bitmap image) {
350             Uri uri = null;
351             try {
352                 File file = new File(getExternalFilesDir(Environment.DIRECTORY_PICTURES), "temp.png");
353                 FileOutputStream fileOutputStream = new FileOutputStream(file);
354                 image.compress(Bitmap.CompressFormat.PNG, 100, fileOutputStream);
355                 fileOutputStream.close();
356                 uri = Uri.fromFile(file);
357             } catch (IOException e) {
358                 Log.d("save bitmap", "IOException: " + e.getMessage());
359             }
360             return uri;
361         }
