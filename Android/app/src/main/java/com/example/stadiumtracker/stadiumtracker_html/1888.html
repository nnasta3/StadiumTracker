117         @Override
118         public boolean onOptionsItemSelected(MenuItem item) {
119             switch (item.getItemId()) {
120                 case R.id.action_logout:
121                     Intent intent = new Intent(this,LoginActivity.class);
122                     startActivity(intent);
123                     return true;
124                 case R.id.action_share:
125                     //Share a string containing username, stadium, number of visits, and most recent visit
126                     String shareString = user.getName()+" has visited "+stadium.getName()+" "+numVisits;
127                     if (numVisits == 1){
128                         shareString += " time ";
129                     }else {
130                         shareString += " times ";
131                     }
132                     shareString += "with the most recent being on "+events.get(0).getDateFullString();
133                     //Can say events[0] is most recent because of sql query
134                     Intent sendIntent = new Intent();
135                     sendIntent.setAction(Intent.ACTION_SEND);
136                     sendIntent.putExtra(Intent.EXTRA_TEXT, shareString);
137                     sendIntent.setType("text/plain");
138                     Intent shareIntent = Intent.createChooser(sendIntent, null);
139                     startActivity(shareIntent);
140                     return true;
141                 default:
142                     //Back button pressed
143                     if (from.equals("visitView")){
144                         Intent backIntent = new Intent(this,VisitViewActivity.class);
145                         backIntent.putExtra("user", user);
146                         backIntent.putExtra("event",getIntent().getSerializableExtra("event"));
147                         backIntent.putExtra("from","visitList");
148                         startActivity(backIntent);
149                     }else {
150                         Intent backIntent = new Intent(this,StadiumsListActivity.class);
151                         backIntent.putExtra("user", user);
152                         startActivity(backIntent);
153                     }
154     
155                     return true;
156     
157             }
158         }

160         public class  ImageFromURL extends AsyncTask<String, Void, Drawable>{
161             @Override
162             protected Drawable doInBackground(String... strings) {
163                 try{
164                     InputStream inputStream = (InputStream) new URL(strings[0]).getContent();
165                     return Drawable.createFromStream(inputStream,"image");
166                 }catch (Exception e){
167                     Log.e("getImage",e.toString());
168                     return null;
169                 }
170             }
171         }
