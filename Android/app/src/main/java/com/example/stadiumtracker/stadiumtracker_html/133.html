320         public boolean onOptionsItemSelected(MenuItem item) {
321             switch (item.getItemId()) {
322                 case R.id.action_logout:
323                     Intent intent = new Intent(this,LoginActivity.class);
324                     startActivity(intent);
325                     return true;
326                 case R.id.action_gps:
327                     //Get gps coords of device
328                     if(!(PackageManager.PERMISSION_GRANTED==checkSelfPermission(Manifest.permission.ACCESS_FINE_LOCATION))){
329                         requestPermissions(LOCATION_PERMS,LOCATION_REQUEST);
330                     }
331                     LocationManager locationManager = (LocationManager) getSystemService(Context.LOCATION_SERVICE);
332                     try{
333                         Location location = locationManager.getLastKnownLocation(LocationManager.GPS_PROVIDER);
334                         //Calculate distance from each stadium in stadiumList
335                         List<Double> dists = new ArrayList<>();
336                         for(int i=0; i<stadiumList.size(); i++){
337                             double x = location.getLatitude() - stadiumList.get(i).getGpsLat();
338                             double y = location.getLongitude() - stadiumList.get(i).getGpsLong();
339                             double dist = Math.sqrt((x*x)+(y*y));
340                             dists.add(dist);
341                         }
342                         //Pick smallest one and set selection to that one
343                         int smallestIndex = getSmallest(dists);
344                         stadiumSpinner.setSelection(smallestIndex);
345                         selectedStadium = stadiumList.get(smallestIndex);
346                     }catch (SecurityException e){
347                         Log.w("Error location","" + e);
348                     }catch (Exception e){
349                         Log.w("Error location","" + e);
350                     }
351                     return true;
352                 default:
353                     //Back button pressed
354                     Intent backIntent = new Intent(this,MainMenuActivity.class);
355                     backIntent.putExtra("user", user);
356                     startActivity(backIntent);
357                     return true;
358     
359             }

466         public boolean validateFields(){
467             //No validation needed for date field, user is not allowed to enter text into that field and the datepicker validates otherwise
468             //Only need to check that the selected stadium is not null or not the hint field
469             if(selectedStadium == null || stadiumSpinner.getSelectedItemPosition()==0){
470                 Toast.makeText(getApplicationContext(),"Please select a stadium",Toast.LENGTH_SHORT).show();
471                 return false;
472             }
473             //validate home team
474             if(selectedHomeTeam == null || homeTeamSpinner.getSelectedItemPosition()==0){
475                 Toast.makeText(getApplicationContext(),"Please select a home team",Toast.LENGTH_SHORT).show();
476                 return false;
477             }
478             //validate road team
479             if(selectedRoadTeam == null || roadTeamSpinner.getSelectedItemPosition()==0){
480                 Toast.makeText(getApplicationContext(),"Please select a road team",Toast.LENGTH_SHORT).show();
481                 return false;
482             }
483             //Make sure home team and road team are different
484             if(selectedHomeTeam.getTeamID() == selectedRoadTeam.getTeamID()){
485                 Toast.makeText(getApplicationContext(),"Selected teams cannot be the same",Toast.LENGTH_SHORT).show();
486                 return false;
487             }
488             //Make sure home team and road team are from same league
489             if(!selectedHomeTeam.getLeague().equals(selectedRoadTeam.getLeague())){
490                 Toast.makeText(getApplicationContext(),"Selected teams must be in the same league",Toast.LENGTH_SHORT).show();
491                 return false;
492             }
493             //Validate home score
494             String homescoreString = homeScore.getText().toString();
495             try{
496                 int homeScoreInt = Integer.parseInt(homescoreString);
497                 if(homeScoreInt < 0){
498                     Toast.makeText(getApplicationContext(),"Home score must be a positive integer",Toast.LENGTH_SHORT).show();
499                     return false;
500                 }
501             }catch(Exception e){
502                 Toast.makeText(getApplicationContext(),"Home score must be a number format",Toast.LENGTH_SHORT).show();
503                 return false;
504             }
505             //Validate road score
506             String roadscoreString = roadScore.getText().toString();
507             try{
508                 int roadScoreInt = Integer.parseInt(roadscoreString);
509                 if(roadScoreInt < 0){
510                     Toast.makeText(getApplicationContext(),"Road score must be a positive integer",Toast.LENGTH_SHORT).show();
511                     return false;
512                 }
513             }catch(Exception e){
514                 Toast.makeText(getApplicationContext(),"Road score must be a number format",Toast.LENGTH_SHORT).show();
515                 return false;
516             }
517             //Validate league
518             if(selectedLeague == null){
519                 Toast.makeText(getApplicationContext(),"Please select a league",Toast.LENGTH_SHORT).show();
520                 return false;
521             }
522             return true;
523         }

547         class  getTeams extends AsyncTask<String, Void, List<Team>> {
548             String ip = getResources().getString(R.string.ip);
549             String port = getResources().getString(R.string.port);
550             String dbName = getResources().getString(R.string.db_name);
551             String user = getResources().getString(R.string.masterUser);
552             String pass = getResources().getString(R.string.masterPass);
553             @Override
554             protected List<Team> doInBackground(String... strings) {
555                 List<Team> out = new ArrayList<>();
556                 try {
557                     // SET CONNECTIONSTRING
558                     Class.forName("net.sourceforge.jtds.jdbc.Driver").newInstance();
559                     Connection DbConn = DriverManager.getConnection("jdbc:jtds:sqlserver://" + ip + ":" + port + "/" + dbName + ";user=" + user + ";password=" + pass);
560                     Statement stmt = DbConn.createStatement();
561                     ResultSet rs = stmt.executeQuery("Select * from [Team] Order By City,Nickname");
562                     while(rs.next()){
563                         int id = rs.getInt(1);
564                         String city = rs.getString(2);
565                         String nickname = rs.getString(3);
566                         String abbrev = rs.getString(4);
567                         int stadiumID = rs.getInt(5);
568                         Stadium stadium = getStadiumById(stadiumID);
569                         String league = rs.getString(6);
570                         out.add(new Team(id,city,nickname,abbrev,stadium,league));
571                     }
572                     DbConn.close();
573                 } catch (Exception e) {
574                     Log.w("Error visit create", "" + e);
575                     return out;
576                 }
577                 return out;
578             }
579         }
